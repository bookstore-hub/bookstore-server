        Note: Pour un côté pratique, j'ai simplement rajouté les tables nécessaires à ce travail à ma base de données
        PostgreSQL existante et déjà déployée dans le cloud. Grâce à ceci, vous n'aurez pas à installer la base
        de données localement. Le nom de schéma "assist" et l'utilisateur "assistadmin" se rapportent à ladite
        base de données et n'ont aucun lien avec ce travail. Pour information, je livre tout de même ci-dessous
        les scripts utilisés.

        // Sequences

        CREATE SEQUENCE assist.book_id_seq
            INCREMENT 50
            START 1
            MINVALUE 1
            MAXVALUE 9223372036854775807
            CACHE 1;

        ALTER SEQUENCE assist.book_id_seq
            OWNER TO assistadmin;

		CREATE SEQUENCE assist.author_id_seq
            INCREMENT 50
            START 1
            MINVALUE 1
            MAXVALUE 9223372036854775807
            CACHE 1;

        ALTER SEQUENCE assist.author_id_seq
            OWNER TO assistadmin;

        // Tables

        CREATE TABLE IF NOT EXISTS assist.book
        (
            id bigint NOT NULL DEFAULT nextval('assist.book_id_seq'::regclass),
            code character varying(30) COLLATE pg_catalog."default" NOT NULL,
            title character varying(50) COLLATE pg_catalog."default" NOT NULL,
            date_of_publication date NOT NULL,
            synopsis character varying(300) COLLATE pg_catalog."default",
            number_of_pages smallint,
            last_modification_date timestamp without time zone,
            last_modification_user character varying(20) COLLATE pg_catalog."default",
            CONSTRAINT book_pkey PRIMARY KEY (id)
        )
        WITH (
            OIDS = FALSE
        )
        TABLESPACE pg_default;

        ALTER TABLE IF EXISTS assist.book
            OWNER to assistadmin;

        CREATE TABLE IF NOT EXISTS assist.author
        (
            id bigint NOT NULL DEFAULT nextval('assist.author_id_seq'::regclass),
            code character varying(30) COLLATE pg_catalog."default" NOT NULL,
            name character varying(30) COLLATE pg_catalog."default" NOT NULL,
            last_modification_date timestamp without time zone,
            last_modification_user character varying(20) COLLATE pg_catalog."default",
            CONSTRAINT author_pkey PRIMARY KEY (id)
        )
        WITH (
            OIDS = FALSE
        )
        TABLESPACE pg_default;

        ALTER TABLE IF EXISTS assist.author
            OWNER to assistadmin;

        CREATE TABLE IF NOT EXISTS assist.book_author
        (
            book_id bigint NOT NULL,
            author_id bigint NOT NULL,
            CONSTRAINT author_ref FOREIGN KEY (author_id)
                REFERENCES assist.author (id) MATCH SIMPLE
                ON UPDATE CASCADE
                ON DELETE CASCADE
                NOT VALID,
            CONSTRAINT book_ref FOREIGN KEY (book_id)
                REFERENCES assist.book (id) MATCH SIMPLE
                ON UPDATE CASCADE
                ON DELETE CASCADE
                NOT VALID
        )
        WITH (
            OIDS = FALSE
        )
        TABLESPACE pg_default;

        ALTER TABLE IF EXISTS assist.book_author
            OWNER to assistadmin;


        CREATE INDEX idx_book_title
        ON assist.book (title);

        CREATE INDEX idx_author_name
        ON assist.author (name);
